package javaexam2.company;


import java.util.HashMap;
import java.util.Map;

public class Company {
    public static void main(String[] args) {

        Map<Integer, Employee> map = new HashMap<Integer, Employee>();

        Secretary secretary = new Secretary("Hilery", 1, "secretary", 800);
        Sales sales = new Sales("Clinten", 2, "sales", 1200);

        map.put(1, secretary);
        map.put(2, sales);

        // for문을 통해 출력
        printAllEmployee(map, false);

        // incentive 지급
        secretary.incentive(100);
        sales.incentive(100);

        printAllEmployee(map, true);



    }

    public static void printAllEmployee(Map<Integer, Employee> map, boolean tax) {
        String column = "name    department     salary";
        StringBuilder sb = new StringBuilder();


        if (tax) {
            column += "      tax";
            sb.append(column).append('\n');
            for (Employee e : map.values()) {

                        sb.append(String.format("%-7s      %s      %s      %s", e.getName(), e.getDepartment(), e.getSalary(), e.tax()))
                        .append('\n');

            }
        } else {
            sb.append(column).append('\n');

            for (Employee e : map.values()) {
                        sb.append(String.format("%-7s      %s      %s", e.getName(), e.getDepartment(), e.getSalary()))
                        .append('\n');

            }
        }

        sb.append('\n');

        System.out.println(sb.toString());
    }
}

package javaexam2.company;

public abstract class Employee {
    private String name;
    private int number;
    private String department;
    private int salary;

    public Employee(){}

    public Employee(String name, int number, String department, int salary) {
        this.name = name;
        this.number = number;
        this.department = department;
        this.salary = salary;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getNumber() {
        return number;
    }

    public void setNumber(int number) {
        this.number = number;
    }

    public String getDepartment() {
        return department;
    }

    public void setDepartment(String department) {
        this.department = department;
    }

    public int getSalary() {
        return salary;
    }

    public void setSalary(int salary) {
        this.salary = salary;
    }

    public abstract double tax();
}

package javaexam2.company;

public class Sales extends Employee implements Bonus{

    public Sales() {}

    public Sales(String name, int number, String department, int salary) {
        super(name, number, department, salary);
    }

    @Override
    public double tax() {
        int originalSalary = super.getSalary();
        return originalSalary * 0.13;
    }

    @Override
    public void incentive(int pay) {

        int incentiveAmount = (int) (pay * 1.2);

        this.setSalary(this.getSalary() + incentiveAmount);
    }


}
