package assignment;

public class Test04 {
    public static void main(String[] args) {
        for(int i = 1; i <= 6; i++) {
            for(int j = 1; j <= 6; j++) {
                for(int k = 1; k <= 6; k++) {
                    int result = i * j * k;

                    if(result % 3 == 0) {
                        System.out.printf("%d*%d*%d = %d\n", i, j, k, result);
                    }
                }
            }
        }
    }
}
