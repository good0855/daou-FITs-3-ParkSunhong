package javaexam2.plane;

public class Airplane extends Plane {
    public Airplane() {}

    public Airplane(String planeName, int fuelSize) {
        super(planeName, fuelSize);
    }

    @Override
    public void flight(int distance){
        int remainFuelSize = super.getFuelSize() - distance * 3;
        setFuelSize(remainFuelSize);
    }
}

package javaexam2.plane;

import java.util.ArrayList;
import java.util.List;

public class PlainTest {

    public static void main(String[] args) {
        // 객체 생성
        Plane airplane = new Airplane("L747", 1000);
        Plane cargoPlane = new CargoPlane("C40", 1000);

        // 리스트 넣기
        List<Plane> planes = new ArrayList<>();
        planes.add(airplane);
        planes.add(cargoPlane);

        // 정보 출력
        printInfo(planes);

        // 100 운항
        int distance = 100;
        airplane.flight(distance);
        cargoPlane.flight(distance);

        System.out.printf("%d 운항\n", distance);
        printInfo(planes);

        // 200 주유
        int fuel = 200;
        airplane.refuel(fuel);
        cargoPlane.refuel(fuel);

        System.out.printf("%d 주유\n", fuel);
        printInfo(planes);



    }
    /*
        비행기 출력 메소드
     */
    public static void printInfo(List<Plane> planes) {
        System.out.println("Planes    fuelSize");
        System.out.println("-------------------");
        for (Plane plane : planes) {
            System.out.printf("%-4s      %s\n", plane.getPlaneName(), plane.getFuelSize());
        }
        System.out.println();
    }
}

package javaexam2.plane;

public abstract class Plane {
    private String planeName;
    private int fuelSize;

    public int getFuelSize(){
        return fuelSize;
    }

    public String getPlaneName() {
        return planeName;
    }

    public void setPlaneName(String planeName) {
        this.planeName = planeName;
    }

    public void setFuelSize(int fuelSize) {
        this.fuelSize = fuelSize;
    }

    public Plane(){}

    public Plane(String planeName, int fuelSize){
        this.planeName = planeName;
        this.fuelSize = fuelSize;
    }

    public void refuel(int fuel) {
        this.fuelSize += fuel;
    }

    public void flight(int distance){}

}
